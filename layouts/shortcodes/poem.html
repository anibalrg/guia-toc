{{- /* Renders a two-column poem with original and translation. @param {string} [title1] Title of the original poem (left column). @param {string} [title2] Title of the translated poem (right column). @param {string} [author] Author of the poem, shown below both columns. @example {{< poem title1="The Rose" title2="La Rosa" author="John Doe" >}} Roses are red,;;Las rosas son rojas, Violets are blue,;;Las violetas son azules, Hugo is great,;;Hugo es genial, And so are you.;;Y tú también. {{< /poem >}} */}} {{- $title1 := .Get "title1" }} {{- $title2 := .Get "title2" }} {{- $author := .Get "author" }} {{- $inner := .Inner | strings.TrimSpace }} {{- $lines := slice }} {{- range $line := split $inner "\n" }} {{- $trimmed := strings.TrimSpace $line }} {{- if $trimmed }} {{- $lines = $lines | append $trimmed }} {{- end }} {{- end -}}
{{- if $title1 }}
{{ $title1 }}
{{- end }}

{{ range $lines }}{{ index (split . ";;") 0 }} {{ end }}

{{- if $title2 }}
{{ $title2 }}
{{- end }}

{{ range $lines }}{{ $parts := split . ";;" }}{{ if gt (len $parts) 1 }}{{ index $parts 1 }}{{ end }} {{ end }}

{{- if $author }}
— {{ $author }}
{{- end }}
