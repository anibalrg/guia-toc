{{- /* Renders a YouTube video thumbnail linking to YouTube. @param {string} [id] The video id. Optional if the id is the first and only positional argument. @param {string} [title] Alt text and aria-label. Defaults to "YouTube video". @param {string} [loading=lazy] The loading attribute of the img element. @param {int} [start] Start time in seconds, passed as &t= in the watch URL. @example {{< yt-thumb 5IUUk5_1Exg >}} @example {{< yt-thumb id=5IUUk5_1Exg title="TOC: Tocados" start=30 >}} */}} {{- $id := or (.Get "id") (.Get 0) }} {{- $title := or (.Get "title") "YouTube video" }} {{- $loading := or (.Get "loading") "lazy" }} {{- $start := .Get "start" }} {{- if not $id }} {{- errorf "The %q shortcode requires an id argument. See %s" .Name .Position }} {{- end }} {{- $watchURL := printf "https://www.youtube.com/watch?v=%s" $id }} {{- if $start }} {{- $watchURL = printf "%s&t=%s" $watchURL ($start | string) }} {{- end }} {{- $thumb := printf "https://img.youtube.com/vi/%s/mqdefault.jpg" $id }} {{- $thumbFallback := printf "https://img.youtube.com/vi/%s/hqdefault.jpg" $id -}}
